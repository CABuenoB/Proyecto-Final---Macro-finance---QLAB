# Conversión de la Colomnua Fechas al formato Correcto
months_map = {
    'Ene.': 'Jan', 'Feb.': 'Feb', 'Mar.': 'Mar', 'Abr.': 'Apr', 'May.': 'May',
    'Jun.': 'Jun', 'Jul.': 'Jul', 'Ago.': 'Aug', 'Sep.': 'Sep', 'Oct.': 'Oct',
    'Nov.': 'Nov', 'Dic.': 'Dec'
}
df['Fecha'] = df['Fecha OG'].replace(months_map, regex=True)

df['Fecha'] = pd.to_datetime(df['Fecha'], format='%b%Y')

# Cáculo del Promedio Móvil (ultimos 12 meses)
df['pbiPromedio'] = df['PBI'].rolling(window=12).mean()

# Identificar Periodo para la proyeccion (ene2015 - feb2020)
start_pre = pd.Timestamp('2015-01-01')
end_pre = pd.Timestamp('2020-02-01')
df['prepandemia'] = df['Fecha'].apply(lambda x: 1 if start_pre <= x <= end_pre else 0)


# Crear la columna "crecimiento" con la variación anual de "pbiPromedio"
df['crecimiento'] = df['pbiPromedio'].pct_change(periods=12, fill_method=None)
